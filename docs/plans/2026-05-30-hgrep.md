# hgrep Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `hgrep`, a harness-agnostic CLI that indexes Claude Code and Pi session transcripts into one SQLite FTS5 store and lets you search across both by keyword plus filters, then resume the matching session in its original harness and directory.

**Architecture:** Per-harness adapters normalize each on-disk JSONL format into a shared `SessionRecord`. An incremental indexer writes records into SQLite (a `sessions` table plus a `sessions_fts` FTS5 table over conversation text only). A search layer runs bm25-ranked FTS queries with metadata filters and emits JSON (default) or a pretty table, each result carrying a ready-to-run resume command. A thin `/recall` Agent Skill drives the CLI.

**Tech Stack:** Python 3.10+, standard library only at runtime (`sqlite3` with FTS5, `json`, `argparse`, `glob`, `shlex`, `datetime`). pytest for tests (dev-only). Packaged via `pyproject.toml` exposing an `hgrep` console script.

`hgrep/harness/base.py`:

```python
from pathlib import Path
from typing import Iterator, Optional, Protocol, Tuple, runtime_checkable

from hgrep.record import SessionRecord

MAX_CONTENT_CHARS = 200_000


@runtime_checkable
class Adapter(Protocol):
    name: str

    def is_available(self) -> bool: ...

    def iter_sessions(self) -> Iterator[Tuple[Path, float, int]]: ...

    def parse(self, path: Path) -> SessionRecord: ...

    def resume_command(self, record: SessionRecord) -> str: ...


def unslug_dir(name: str) -> str:
    """Recover an absolute path from a dash-slug directory name.

    Both harnesses encode the cwd as the directory name with '/' replaced by
    '-'. This is a best-effort fallback only; the real cwd is read from the
    file contents when present.
    """
    return "/" + name.strip("-").replace("-", "/")
```

```python
import json
import os
import shlex
from pathlib import Path
from typing import Iterator, Optional, Tuple

from hgrep.harness.base import MAX_CONTENT_CHARS, unslug_dir
from hgrep.record import SessionRecord


class ClaudeCodeAdapter:
    name = "claude-code"

    def __init__(self, root: Optional[Path] = None):
        self.root = Path(root) if root else Path.home() / ".claude" / "projects"

    def is_available(self) -> bool:
        return self.root.is_dir()

    def iter_sessions(self) -> Iterator[Tuple[Path, float, int]]:
        if not self.root.is_dir():
            return
        for path in self.root.glob("*/*.jsonl"):
            try:
                st = path.stat()
            except OSError:
                continue
            yield path, st.st_mtime, st.st_size

    def parse(self, path: Path) -> SessionRecord:
        cwd = None
        branch = None
        custom = None
        ai = None
        first_user = None
        texts = []
        ts_min = None
        ts_max = None
        msg_count = 0

        for line in path.read_text(errors="replace").splitlines():
            line = line.strip()
            if not line:
                continue
            try:
                o = json.loads(line)
            except json.JSONDecodeError:
                continue
            if o.get("cwd") and cwd is None:
                cwd = o["cwd"]
            if o.get("gitBranch") and branch is None:
                branch = o["gitBranch"]
            if o.get("customTitle"):
                custom = o["customTitle"]
            if o.get("aiTitle"):
                ai = o["aiTitle"]
            ts = o.get("timestamp")
            if ts:
                if ts_min is None or ts < ts_min:
                    ts_min = ts
                if ts_max is None or ts > ts_max:
                    ts_max = ts
            t = o.get("type")
            if t not in ("user", "assistant"):
                continue
            msg_count += 1
            content = o.get("message", {}).get("content")
            if t == "user":
                if isinstance(content, str):
                    texts.append(content)
                    if first_user is None:
                        first_user = content
                elif isinstance(content, list):
                    for p in content:
                        if isinstance(p, dict) and p.get("type") == "text":
                            txt = p.get("text", "")
                            texts.append(txt)
                            if first_user is None:
                                first_user = txt
            elif isinstance(content, list):  # assistant
                for p in content:
                    if isinstance(p, dict) and p.get("type") == "text":
                        texts.append(p.get("text", ""))

        session_id = path.stem
        if cwd is None:
            cwd = unslug_dir(path.parent.name)
        first_user = (first_user or "").strip()
        title = (custom or ai or first_user or "(untitled)").strip()[:200] or "(untitled)"
        content_text = "\n".join(texts)[:MAX_CONTENT_CHARS]
        st = path.stat()
        return SessionRecord(
            harness=self.name,
            session_id=session_id,
            cwd=cwd,
            project=os.path.basename(cwd.rstrip("/")) or cwd,
            git_branch=branch,
            title=title,
            first_user_msg=first_user[:500],
            content_text=content_text,
            started_at=ts_min or "",
            last_activity_at=ts_max or ts_min or "",
            msg_count=msg_count,
            file_path=str(path),
            file_mtime=st.st_mtime,
            file_size=st.st_size,
        )

    def resume_command(self, record: SessionRecord) -> str:
        return f"(cd {shlex.quote(record.cwd)} && claude --resume {record.session_id})"
```

```python
import json
import os
import shlex
from pathlib import Path
from typing import Iterator, Optional, Tuple

from hgrep.harness.base import MAX_CONTENT_CHARS, unslug_dir
from hgrep.record import SessionRecord


class PiAdapter:
    name = "pi"

    def __init__(self, root: Optional[Path] = None):
        self.root = Path(root) if root else Path.home() / ".pi" / "agent" / "sessions"

    def is_available(self) -> bool:
        return self.root.is_dir()

    def iter_sessions(self) -> Iterator[Tuple[Path, float, int]]:
        if not self.root.is_dir():
            return
        for path in self.root.glob("*/*.jsonl"):
            try:
                st = path.stat()
            except OSError:
                continue
            yield path, st.st_mtime, st.st_size

    def parse(self, path: Path) -> SessionRecord:
        sid = None
        cwd = None
        name = None
        first_user = None
        texts = []
        ts_min = None
        ts_max = None
        msg_count = 0

        for line in path.read_text(errors="replace").splitlines():
            line = line.strip()
            if not line:
                continue
            try:
                o = json.loads(line)
            except json.JSONDecodeError:
                continue
            ts = o.get("timestamp")
            if ts:
                if ts_min is None or ts < ts_min:
                    ts_min = ts
                if ts_max is None or ts > ts_max:
                    ts_max = ts
            typ = o.get("type")
            if typ == "session":
                sid = o.get("id")
                cwd = o.get("cwd")
            elif typ == "session_info" and o.get("name"):
                name = o["name"]
            elif typ == "message":
                m = o.get("message", {})
                role = m.get("role")
                content = m.get("content")
                if role not in ("user", "assistant"):
                    continue
                msg_count += 1
                if isinstance(content, list):
                    for p in content:
                        if isinstance(p, dict) and p.get("type") == "text":
                            txt = p.get("text", "")
                            texts.append(txt)
                            if role == "user" and first_user is None:
                                first_user = txt

        if sid is None:
            stem = path.stem
            sid = stem.split("_", 1)[1] if "_" in stem else stem
        if cwd is None:
            cwd = unslug_dir(path.parent.name)
        first_user = (first_user or "").strip()
        title = (name or first_user or "(untitled)").strip()[:200] or "(untitled)"
        content_text = "\n".join(texts)[:MAX_CONTENT_CHARS]
        st = path.stat()
        return SessionRecord(
            harness=self.name,
            session_id=sid,
            cwd=cwd,
            project=os.path.basename(cwd.rstrip("/")) or cwd,
            git_branch=None,
            title=title,
            first_user_msg=first_user[:500],
            content_text=content_text,
            started_at=ts_min or "",
            last_activity_at=ts_max or ts_min or "",
            msg_count=msg_count,
            file_path=str(path),
            file_mtime=st.st_mtime,
            file_size=st.st_size,
        )

    def resume_command(self, record: SessionRecord) -> str:
        return f"(cd {shlex.quote(record.cwd)} && pi --session {record.session_id})"
```

```python
import os
import sqlite3
from pathlib import Path
from typing import Iterable, Optional

from hgrep.harness.base import Adapter

SCHEMA = """
CREATE TABLE IF NOT EXISTS sessions (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  harness TEXT NOT NULL,
  session_id TEXT NOT NULL,
  cwd TEXT,
  project TEXT,
  git_branch TEXT,
  title TEXT,
  first_user_msg TEXT,
  started_at TEXT,
  last_activity_at TEXT,
  msg_count INTEGER,
  resume_command TEXT,
  file_path TEXT NOT NULL UNIQUE,
  file_mtime REAL,
  file_size INTEGER
);
CREATE INDEX IF NOT EXISTS idx_sessions_harness ON sessions(harness);
CREATE VIRTUAL TABLE IF NOT EXISTS sessions_fts
  USING fts5(title, first_user_msg, content);
"""


def default_db_path() -> Path:
    env = os.environ.get("HGREP_DB")
    if env:
        return Path(env)
    return Path.home() / ".cache" / "hgrep" / "index.db"


def connect(db_path: Optional[Path] = None) -> sqlite3.Connection:
    path = Path(db_path) if db_path else default_db_path()
    path.parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(str(path))
    conn.row_factory = sqlite3.Row
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA busy_timeout=5000")
    conn.executescript(SCHEMA)
    return conn


def rebuild(conn: sqlite3.Connection) -> None:
    conn.executescript(
        "DROP TABLE IF EXISTS sessions; DROP TABLE IF EXISTS sessions_fts;"
    )
    conn.executescript(SCHEMA)
    conn.commit()


def _upsert(conn: sqlite3.Connection, rec, resume_command: str) -> None:
    row = conn.execute(
        "SELECT id FROM sessions WHERE file_path=?", (rec.file_path,)
    ).fetchone()
    cols = (
        rec.harness, rec.session_id, rec.cwd, rec.project, rec.git_branch,
        rec.title, rec.first_user_msg, rec.started_at, rec.last_activity_at,
        rec.msg_count, resume_command, rec.file_path, rec.file_mtime, rec.file_size,
    )
    if row:
        rid = row["id"]
        conn.execute(
            """UPDATE sessions SET harness=?,session_id=?,cwd=?,project=?,git_branch=?,
               title=?,first_user_msg=?,started_at=?,last_activity_at=?,msg_count=?,
               resume_command=?,file_path=?,file_mtime=?,file_size=? WHERE id=?""",
            cols + (rid,),
        )
        conn.execute("DELETE FROM sessions_fts WHERE rowid=?", (rid,))
    else:
        cur = conn.execute(
            """INSERT INTO sessions(harness,session_id,cwd,project,git_branch,title,
               first_user_msg,started_at,last_activity_at,msg_count,resume_command,
               file_path,file_mtime,file_size)
               VALUES(?,?,?,?,?,?,?,?,?,?,?,?,?,?)""",
            cols,
        )
        rid = cur.lastrowid
    conn.execute(
        "INSERT INTO sessions_fts(rowid,title,first_user_msg,content) VALUES(?,?,?,?)",
        (rid, rec.title, rec.first_user_msg, rec.content_text),
    )


def refresh(conn: sqlite3.Connection, adapters: Iterable[Adapter]) -> dict:
    stats = {"parsed": 0, "skipped": 0, "deleted": 0}
    for adapter in adapters:
        if not adapter.is_available():
            continue
        seen = set()
        for path, mtime, size in adapter.iter_sessions():
            fp = str(path)
            seen.add(fp)
            row = conn.execute(
                "SELECT file_mtime,file_size FROM sessions WHERE file_path=?", (fp,)
            ).fetchone()
            if row and abs(row["file_mtime"] - mtime) < 1e-6 and row["file_size"] == size:
                stats["skipped"] += 1
                continue
            try:
                rec = adapter.parse(path)
                cmd = adapter.resume_command(rec)
            except Exception:
                continue
            _upsert(conn, rec, cmd)
            stats["parsed"] += 1
        stored = conn.execute(
            "SELECT id,file_path FROM sessions WHERE harness=?", (adapter.name,)
        ).fetchall()
        for r in stored:
            if r["file_path"] not in seen:
                conn.execute("DELETE FROM sessions_fts WHERE rowid=?", (r["id"],))
                conn.execute("DELETE FROM sessions WHERE id=?", (r["id"],))
                stats["deleted"] += 1
    conn.commit()
    return stats
```

`hgrep/search.py`:

```python
import sqlite3
from datetime import datetime, timedelta, timezone
from typing import List, Optional

_PUBLIC_FIELDS = [
    "harness", "session_id", "title", "project", "cwd", "git_branch",
    "last_activity_at", "msg_count", "snippet", "resume_command",
]


def build_match(query: str) -> str:
    terms = [t for t in query.split() if t]
    return " ".join('"' + t.replace('"', '""') + '"' for t in terms)


def parse_since(value: str) -> str:
    v = value.strip().lower()
    units = {"d": 1, "w": 7}
    if len(v) >= 2 and v[-1] in units and v[:-1].isdigit():
        days = int(v[:-1]) * units[v[-1]]
        dt = datetime.now(timezone.utc) - timedelta(days=days)
        return dt.strftime("%Y-%m-%dT%H:%M:%S.000Z")
    return value


def search(
    conn: sqlite3.Connection,
    query: str,
    *,
    since: Optional[str] = None,
    until: Optional[str] = None,
    project: Optional[str] = None,
    branch: Optional[str] = None,
    harness: Optional[str] = None,
    limit: int = 20,
) -> List[dict]:
    match = build_match(query)
    if not match:
        return []
    where = ["sessions_fts MATCH ?"]
    params: list = [match]
    if since:
        where.append("s.last_activity_at >= ?")
        params.append(parse_since(since))
    if until:
        where.append("s.last_activity_at <= ?")
        params.append(until)
    if project:
        where.append("s.project LIKE ?")
        params.append(f"%{project}%")
    if branch:
        where.append("s.git_branch = ?")
        params.append(branch)
    if harness:
        where.append("s.harness = ?")
        params.append(harness)
    sql = f"""
        SELECT s.*, bm25(sessions_fts) AS rank,
               snippet(sessions_fts, 2, '[', ']', '...', 12) AS snippet
        FROM sessions_fts JOIN sessions s ON s.id = sessions_fts.rowid
        WHERE {' AND '.join(where)}
        ORDER BY rank ASC, s.last_activity_at DESC
        LIMIT ?
    """
    params.append(limit)
    return [dict(r) for r in conn.execute(sql, params).fetchall()]


def to_public(results: List[dict]) -> List[dict]:
    return [{k: r.get(k) for k in _PUBLIC_FIELDS} for r in results]


def format_pretty(results: List[dict]) -> str:
    if not results:
        return "(no matches)"
    lines = []
    for r in results:
        proj = r.get("project") or ""
        if r.get("git_branch"):
            proj = f"{proj}/{r['git_branch']}"
        lines.append(
            f"{r.get('harness'):<11} {r.get('last_activity_at', '')[:10]}  "
            f"{(r.get('title') or '')[:50]:<50}  {proj}"
        )
        snip = (r.get("snippet") or "").replace("\n", " ")
        if snip:
            lines.append(f"    {snip}")
        lines.append(f"    {r.get('resume_command')}")
    return "\n".join(lines)
```

```python
import argparse
import json
import sys
from typing import List, Optional

from hgrep.index import connect, refresh, rebuild, default_db_path
from hgrep.registry import default_adapters
from hgrep.search import search, to_public, format_pretty

_SUBCOMMANDS = {"search", "reindex", "stats"}


def _build_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(prog="hgrep")
    sub = parser.add_subparsers(dest="cmd", required=True)

    s = sub.add_parser("search", help="search sessions across harnesses")
    s.add_argument("query")
    s.add_argument("--since")
    s.add_argument("--until")
    s.add_argument("--project")
    s.add_argument("--branch")
    s.add_argument("--harness")
    s.add_argument("--limit", type=int, default=20)
    s.add_argument("--pretty", action="store_true")
    s.add_argument("--no-refresh", action="store_true")

    r = sub.add_parser("reindex", help="refresh the index")
    r.add_argument("--rebuild", action="store_true")

    sub.add_parser("stats", help="show index stats")
    return parser


def _gather_stats(conn) -> dict:
    rows = conn.execute(
        "SELECT harness, COUNT(*) AS n FROM sessions GROUP BY harness"
    ).fetchall()
    by_harness = {row["harness"]: row["n"] for row in rows}
    total = conn.execute("SELECT COUNT(*) FROM sessions").fetchone()[0]
    return {"total": total, "by_harness": by_harness, "db": str(default_db_path())}


def main(argv: Optional[List[str]] = None, adapters=None) -> int:
    argv = list(sys.argv[1:] if argv is None else argv)
    if argv and argv[0] not in _SUBCOMMANDS and not argv[0].startswith("-"):
        argv = ["search"] + argv

    args = _build_parser().parse_args(argv)
    adapters = adapters if adapters is not None else default_adapters()
    conn = connect()

    if args.cmd == "search":
        if not args.no_refresh:
            refresh(conn, adapters)
        results = search(
            conn,
            args.query,
            since=args.since,
            until=args.until,
            project=args.project,
            branch=args.branch,
            harness=args.harness,
            limit=args.limit,
        )
        if args.pretty:
            print(format_pretty(results))
        else:
            print(json.dumps(to_public(results), indent=2))
    elif args.cmd == "reindex":
        if args.rebuild:
            rebuild(conn)
        print(json.dumps(refresh(conn, adapters)))
    elif args.cmd == "stats":
        print(json.dumps(_gather_stats(conn)))
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```
